cmake_minimum_required(VERSION 3.20)
project(wenlock2-server LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Generate config.h needed by libcaptcha
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/config.h" "/* Generated by CMake */\n#define HAVE_STDLIB_H 1\n")

# Set ASIO_INCLUDE_DIR so that Crow's Findasio can locate it
set(ASIO_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/extern/asio/include" CACHE PATH "Path to asio headers" FORCE)

# Build libcaptcha as a static C library
set(LIBCAPTCHA_SOURCES
    extern/libcaptcha/src/string.c
    extern/libcaptcha/src/pattern.c
    extern/libcaptcha/src/image.c
    extern/libcaptcha/src/font.c
    extern/libcaptcha/src/generator1.c
    extern/libcaptcha/src/pattern-gen.c
    extern/libcaptcha/src/transform.c
    extern/libcaptcha/src/libcaptcha.c
    extern/libcaptcha/src/utf8.c
    extern/libcaptcha/src/preset0.c
    extern/libcaptcha/src/utils.c
)

add_library(captcha STATIC ${LIBCAPTCHA_SOURCES})
target_include_directories(captcha PUBLIC 
    extern/libcaptcha/src
    ${CMAKE_CURRENT_BINARY_DIR}
)
target_link_libraries(captcha PRIVATE m)

# Crow configuration
set(CROW_BUILD_EXAMPLES OFF CACHE BOOL "Build Crow examples" FORCE)
set(CROW_BUILD_TESTS OFF CACHE BOOL "Build Crow tests" FORCE)
add_subdirectory(extern/Crow)

# Build main server executable
add_executable(wenlock2-server src/main.cpp)
target_include_directories(wenlock2-server PRIVATE 
    include
    extern/libcaptcha/src
    extern/asio/include
)
target_link_libraries(wenlock2-server PRIVATE 
    Crow::Crow
    captcha
    pthread
)
